# DAY 3-4 deliverable: src/Engine.jl

function run!(sim::Simulation)::Results
    p = sim.project
    
    # Initialize (calls functions from multiple modules)
    Input.validate(p)
    TopoSort.sort_network!(p)
    DwFlow.compute_initial_flows!(p)
    
    results = Results(p)
    
    while p.current_time < p.end_time
        dt = p.routing_step
        
        # Hydrology (each function from its own module)
        Rain.update!(p)
        for sc in p.subcatchments
            Runoff.compute!(sc, dt)
            Infil.compute!(sc, dt)
            if p.has_groundwater
                GWater.compute!(sc, dt)
            end
            if p.has_snow
                Snow.compute!(sc, dt)
            end
        end
        
        # Hydraulics
        if p.route_model == DYNWAVE
            Routing.dynwave_route!(p, dt)
        else
            KinWave.route!(p, dt)
        end
        
        # Controls
        Controls.evaluate!(p)
        
        # Mass balance
        MassBal.update!(p, dt)
        
        # Record
        if is_report_time(p)
            record!(results, p)
        end
        
        p.current_time += dt
    end
    
    Stats.compute!(results)
    return results
end

### The Key Insight: Only the Inner Loop Goes on GPU

```
DON'T port all 50 modules to GPU. Only the HOT LOOP:

  CPU handles:                    GPU handles:
  ├── INP parsing                 ├── Dynamic wave routing
  ├── Initialization              │   ├── All conduit flows (parallel)
  ├── Rainfall updates            │   ├── All node depths (parallel)
  ├── Control rule evaluation     │   └── Friction slope computation
  ├── Output writing              │
  ├── Mass balance                └── (That's it. 3 kernels.)
  └── Everything else             
  
  The dynwave step is 90%+ of runtime for large models.
  GPU-accelerate ONLY that and get 90% of the speedup.
```

```
DAY 3-5: GPU memory layout

  THE CRITICAL DESIGN DECISION:
  
  CPU memory (Array of Structs):
    links[0] = {flow=5.2, depth=1.3, area=2.1, roughness=0.013, ...}
    links[1] = {flow=3.8, depth=0.9, area=1.5, roughness=0.015, ...}
    links[2] = {flow=7.1, depth=1.8, area=3.2, roughness=0.013, ...}
  
  GPU memory (Struct of Arrays — MUCH faster for GPU):
    flows     = [5.2, 3.8, 7.1, ...]    ← contiguous memory
    depths    = [1.3, 0.9, 1.8, ...]    ← contiguous memory
    areas     = [2.1, 1.5, 3.2, ...]    ← contiguous memory
    roughness = [0.013, 0.015, 0.013, ...] ← contiguous memory
  
  GPU reads entire array in ONE memory transaction.
  This is why GPU SWMM5 can be 100-1000× faster.
```
```cpp
// DAY 3 deliverable: gpu_kernels/dynwave_kernels.cu

#include <cuda_runtime.h>

// Device arrays (GPU memory)
struct DeviceNetwork {
    // Link arrays (Struct of Arrays layout)
    double* d_link_flows;
    double* d_link_old_flows;
    double* d_link_areas;
    double* d_link_depths;
    double* d_link_roughness;
    double* d_link_lengths;
    double* d_link_widths;
    int*    d_link_from_node;
    int*    d_link_to_node;
    int     n_links;
    
    // Node arrays
    double* d_node_heads;
    double* d_node_depths;
    double* d_node_volumes;
    double* d_node_inflows;
    double* d_node_areas;      // surface area
    int     n_nodes;
};

// Kernel 1: Compute friction slope for ALL conduits simultaneously
__global__ void compute_friction_slopes(
    const double* __restrict__ flows,
    const double* __restrict__ areas,
    const double* __restrict__ roughness,
    const double* __restrict__ widths,
    double* __restrict__ friction_slopes,
    int n_links)
{
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= n_links) return;
    
    double Q = flows[i];
    double A = areas[i];
    
    if (A <= 0.0 || Q == 0.0) {
        friction_slopes[i] = 0.0;
        return;
    }
    
    double hRadius = A / widths[i];
    double sf = roughness[i] * Q / (A * pow(hRadius, 2.0/3.0));
    sf = sf * sf;
    
    friction_slopes[i] = (Q < 0.0) ? -sf : sf;
}

// Kernel 2: Update flows for ALL conduits simultaneously
__global__ void update_link_flows(
    const double* __restrict__ old_flows,
    const double* __restrict__ areas,
    const double* __restrict__ lengths,
    const double* __restrict__ friction_slopes,
    const double* __restrict__ node_heads,
    const int* __restrict__ from_nodes,
    const int* __restrict__ to_nodes,
    double* __restrict__ new_flows,
    double dt, double gravity,
    int n_links)
{
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= n_links) return;
    
    double A = areas[i];
    if (A <= 0.0) { new_flows[i] = 0.0; return; }
    
    double hUp = node_heads[from_nodes[i]];
    double hDown = node_heads[to_nodes[i]];
    double dhdx = (hUp - hDown) / lengths[i];
    
    double Qnew = old_flows[i] + dt * gravity * A * (dhdx - friction_slopes[i]);
    
    // Clamp
    double Qmax = A * 50.0;
    new_flows[i] = fmin(fmax(Qnew, -Qmax), Qmax);
}

// Kernel 3: Update node depths from link flows
__global__ void update_node_depths(
    double* __restrict__ depths,
    double* __restrict__ heads,
    double* __restrict__ volumes,
    const double* __restrict__ inflows,
    const double* __restrict__ surface_areas,
    const double* __restrict__ invert_elevs,
    const double* __restrict__ max_depths,
    double dt, int n_nodes)
{
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= n_nodes) return;
    
    double dV = inflows[i] * dt;
    volumes[i] += dV;
    
    double A_surf = fmax(surface_areas[i], 1e-6);
    depths[i] = fmax(volumes[i] / A_surf, 0.0);
    depths[i] = fmin(depths[i], max_depths[i]);
    heads[i] = invert_elevs[i] + depths[i];
}

// Host function: one dynwave timestep
void gpu_dynwave_step(DeviceNetwork& net, double dt) {
    int block_size = 256;
    int link_blocks = (net.n_links + block_size - 1) / block_size;
    int node_blocks = (net.n_nodes + block_size - 1) / block_size;
    
    // Step 1: friction slopes (all links parallel)
    compute_friction_slopes<<<link_blocks, block_size>>>(
        net.d_link_flows, net.d_link_areas, 
        net.d_link_roughness, net.d_link_widths,
        net.d_friction_slopes, net.n_links);
    
    // Step 2: update flows (all links parallel)
    update_link_flows<<<link_blocks, block_size>>>(
        net.d_link_old_flows, net.d_link_areas,
        net.d_link_lengths, net.d_friction_slopes,
        net.d_node_heads, net.d_link_from_node,
        net.d_link_to_node, net.d_link_flows,
        dt, GRAVITY, net.n_links);
    
    // Step 3: accumulate link flows at nodes (needs atomic adds)
    // ... (node inflow summation kernel)
    
    // Step 4: update node depths (all nodes parallel)  
    update_node_depths<<<node_blocks, block_size>>>(
        net.d_node_depths, net.d_node_heads,
        net.d_node_volumes, net.d_node_inflows,
        net.d_node_areas, net.d_invert_elevs,
        net.d_max_depths, dt, net.n_nodes);
}

            project.subcatchments(i) = hydrology.runoff(...
                project.subcatchments(i), dt);
            project.subcatchments(i) = hydrology.infil_horton(...
                project.subcatchments(i), dt);
        end
        
        % Hydraulics (THE INTERESTING PART — set breakpoint here!)
        if project.route_model == 1  % DYNWAVE
            project = hydraulics.routing(project, dt);
        else
            project = hydraulics.kinwave(project, dt);
        end
        
        % Record results at report steps
        if mod(step_count, project.report_ratio) == 0
            results.time(end+1) = project.current_time;
            results.node_depths(end+1,:) = [project.nodes.depth];
            results.link_flows(end+1,:) = [project.links.flow];
        end
        
        % Progress
        step_count = step_count + 1;
        if mod(step_count, 100) == 0
            fprintf('  %.1f%% complete\n', ...
                100 * (project.current_time - project.start_time) / ...
                (project.end_time - project.start_time));
        end
        
        project.current_time = project.current_time + dt;
    end
    
    % Phase 3: Finalize
    results.stats = output.statistics(results, project);
    results.mass_balance = output.mass_balance(project);
    results.project = project;
    
    fprintf('Complete. Continuity error: %.3f%%\n', ...
        results.mass_balance.continuity_error);
end
```
