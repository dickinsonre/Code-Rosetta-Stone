```ruby
        theta = 2.0 * Math.acos(1.0 - 2.0 * y / d)

        # Same geometry as circular
        d = @geom1
        y = [y, d].min
        theta = 2.0 * Math.acos(1.0 - 2.0 * y / d)

        0.0

      else
        0.0
      end
```

```ruby
# lib/swmm5/hydraulics/routing.rb
# The Saint-Venant equations, in Ruby.

module SWMM5
  module Hydraulics
    module Routing
      GRAVITY = 32.2  # ft/s² (9.81 for metric)
      
      # Compute friction slope using Manning's equation
      # This is the same algorithm as routing.c in the Rosetta Stone
      def self.friction_slope(link, flow, area)
        return 0.0 if area <= 0.0 || flow == 0.0
        
        h_radius = area / link.xsect.width_at_depth(link.depth)
        return 0.0 if h_radius <= 0.0
        
        sf = link.roughness * flow / (area * h_radius**(2.0/3.0))
        sf = sf * sf
        
        flow < 0 ? -sf : sf
      end
      
      # Compute new flow using explicit finite-difference
      # Saint-Venant momentum equation
      def self.compute_flow(link, nodes, dt)
        area = link.xsect.area_at_depth(link.depth)
        return 0.0 if area <= 0.0
        
        h_up = nodes[link.from_node].head
        h_down = nodes[link.to_node].head
        
        dhdx = (h_up - h_down) / link.length
        sf = friction_slope(link, link.old_flow, area)
        
        q_new = link.old_flow + dt * GRAVITY * area * (dhdx - sf)
        
        # Velocity cap: prevent unrealistic flows
        q_max = area * 50.0
        q_new.clamp(-q_max, q_max)
      end
      
      # Route all links for one time step
      def self.dynwave_step(project, dt)
        # Save old state
        project.links.each do |link|
          link.old_flow = link.flow
          link.old_depth = link.depth
          link.old_volume = link.volume
        end
        
        # Compute new flows for all links
        project.links.each do |link|
          next unless link.conduit?
          link.flow = compute_flow(link, project.nodes, dt)
          
          # Update depth from flow
          update_link_depth(link)
        end
        
        # Update node depths from link flows
        project.nodes.each_with_index do |node, i|
          next if node.outfall?
          update_node_depth(node, project.links, dt)
        end
      end
      
      # Force main routing — Hazen-Williams when pressurized
      def self.force_main_friction(link, flow, area)
        return friction_slope(link, flow, area) unless link.xsect.shape == :force_main
        
        # When full, use Hazen-Williams
        if link.depth >= link.xsect.geom1  # full pipe
          c_factor = link.xsect.geom2      # H-W C-factor
          d = link.xsect.geom1
          # Hazen-Williams: V = 0.849 * C * R^0.63 * S^0.54
          # Rearranged for friction slope
          v = flow / area
          r = area / (Math::PI * d)
          sf = (v / (0.849 * c_factor * r**0.63))**(1.0/0.54)
          flow < 0 ? -sf : sf
        else
          # Partial flow: use Manning's
          friction_slope(link, flow, area)
        end
      end
      
      private
      
      def self.update_link_depth(link)
        # Iterative depth-from-flow using cross-section geometry
        # Newton-Raphson to find depth that gives current flow area
        target_area = link.flow.abs / [link.velocity.abs, 0.01].max
        
        y = link.depth  # initial guess
        3.times do
          a = link.xsect.area_at_depth(y)
          w = link.xsect.width_at_depth(y)
          break if w <= 0
          y -= (a - target_area) / w
          y = y.clamp(0.0, link.xsect.geom1)
        end
        
        link.depth = y
      end
      
      def self.update_node_depth(node, links, dt)
        # Sum inflows and outflows
        total_in = node.lateral_inflow
        total_out = 0.0
        
        node.inlinks.each { |i| total_in += links[i].flow.abs }
        node.outlinks.each { |i| total_out += links[i].flow.abs }
        
        node.total_inflow = total_in
        
        # Update volume and depth
        dv = (total_in - total_out) * dt
        node.volume = [node.volume + dv, 0.0].max
        
        # Simple area-based depth (storage nodes use stage-area curve)
        surface_area = [node.max_depth * 1.0, 1e-6].max  # simplified
        node.depth = [node.volume / surface_area, 0.0].max
        node.depth = [node.depth, node.max_depth].min
        node.head = node.invert_elev + node.depth
        
        # Track overflow
        if node.depth >= node.max_depth
          node.overflow = (total_in - total_out).clamp(0.0, Float::INFINITY)
        else
          node.overflow = 0.0
        end
      end
    end
  end
end
```

```ruby
# lib/swmm5/engine.rb

module SWMM5
  class Simulation
    attr_reader :project, :results, :current_time
    
    def initialize(inp_file, **options)
      inp_content = File.read(inp_file)
      @project = Parser::InpParser.parse(inp_content)
      
      # Apply parameter overrides
      options.each do |key, value|
        case key
        when :manning_n
          @project.links.each { |l| l.roughness = value if l.conduit? }
        when :imperv
          @project.subcatchments.each { |s| s.imperv = value }
        end
      end
      
      @started = false
      @step_count = 0
    end
    
    # Full run — returns Results object
    def run(&block)
      initialize_simulation
      results = Results.new(@project)
      
      while @project.current_time < @project.end_time
        step
        
        if report_time?
          results.record(@project)
        end
        
        # Yield to block for real-time control
        yield self if block_given?
        
        @project.advance_time
      end
      
      finalize_simulation
      results.compute_statistics
      results
    end
    
    # Step-by-step — for real-time control (PySWMM-style)
    def each_step
      return enum_for(:each_step) unless block_given?
      
      initialize_simulation
      
      while @project.current_time < @project.end_time
        step
        yield self
        @project.advance_time
      end
      
      finalize_simulation
    end
    
    # Accessors for real-time control
    def node(id)
      idx = @project.node_index[id]
      raise KeyError, "Node '#{id}' not found" unless idx
      @project.nodes[idx]
    end
    
    def link(id)
      idx = @project.link_index[id]
      raise KeyError, "Link '#{id}' not found" unless idx
      @project.links[idx]
    end
    
    def percent_complete
      total = @project.end_time - @project.start_time
      elapsed = @project.current_time - @project.start_time
      total > 0 ? (elapsed / total * 100.0) : 0.0
    end
    
    private
    
    def initialize_simulation
      @project.validate!
      Hydraulics::TopoSort.sort!(@project)
      Hydraulics::DwFlow.compute_initial_flows!(@project)
      @started = true
    end
    
    def step
      dt = @project.routing_step
      
      # Hydrology
      Hydrology::Rainfall.update!(@project)
      @project.subcatchments.each do |sc|
        Hydrology::Subcatchment.compute_runoff!(sc, dt)
        Hydrology::Infiltration.compute!(sc, dt)
        Hydrology::Groundwater.compute!(sc, dt) if @project.groundwater?
        Hydrology::Snowmelt.compute!(sc, dt) if @project.snow?
      end
      
      # Hydraulics
      case @project.route_model
      when :dynwave
        Hydraulics::Routing.dynwave_step(@project, dt)
      when :kinwave
        Hydraulics::KinWave.route!(@project, dt)
      end
      
      # Controls
      Controls::RuleEvaluator.evaluate!(@project) if @project.controls?
      
      # Mass balance
      Output::MassBalance.update!(@project, dt)
      
      @step_count += 1
    end
    
    def report_time?
      @step_count % @project.report_ratio == 0
    end
    
    def finalize_simulation
      @started = false
    end
  end
end
```

```ruby
# lib/swmm5/icm/network_bridge.rb
# Bridge between ICM InfoWorks network and SWMM5 engine

module SWMM5
  module ICM
    class NetworkBridge
      # Convert ICM network selection to SWMM5 Project
      def self.from_icm_network(icm_net, selection = nil)
        project = SWMM5::Project.new
        
        # Import junctions
        icm_net.row_objects('cams_manhole').each do |mh|
          next if selection && !selection.include?(mh.id)
          project.add_node(Node.new(
            id: mh.id,
            node_type: :junction,
            invert_elev: mh['chamber_floor'],
            max_depth: mh['shaft_depth'] || mh['ground_level'] - mh['chamber_floor']
          ))
        end
        
        # Import conduits
        icm_net.row_objects('cams_pipe').each do |pipe|
          next if selection && !selection.include?(pipe.id)
          project.add_link(Link.new(
            id: pipe.id,
            link_type: :conduit,
            from_node: pipe['us_node_id'],
            to_node: pipe['ds_node_id'],
            length: pipe['conduit_length'],
            roughness: pipe['roughness_N'] || 0.013,
            xsect: CrossSection.new(
              shape: icm_shape_to_swmm(pipe['shape']),
              geom1: pipe['conduit_height'] / 1000.0,
              geom2: pipe['conduit_width'] / 1000.0
            ),
            in_offset: pipe['us_invert'] - node_invert(pipe['us_node_id']),
            out_offset: pipe['ds_invert'] - node_invert(pipe['ds_node_id'])
          ))
        end
        
        # Import subcatchments
        icm_net.row_objects('cams_subcatchment').each do |sc|
          next if selection && !selection.include?(sc.id)
          project.add_subcatchment(Subcatchment.new(
            id: sc.id,
            area: sc['total_area'] / 10000.0,  # m² → ha
            imperv: sc['percentage_impermeable'],
            width: sc['runoff_routing_value'],
            slope: sc['ground_slope'],
            outlet: sc['node_id']
          ))
        end
        
        project
      end
      
      # Write SWMM5 results back to ICM user fields
      def self.write_results_to_icm(icm_net, results)
        icm_net.transaction_begin
        
        icm_net.row_objects('cams_manhole').each do |mh|
          swmm_node = results.node(mh.id)
          next unless swmm_node
          
          mh['user_number_1'] = swmm_node.peak_depth
          mh['user_number_2'] = swmm_node.peak_inflow
          mh['user_number_3'] = swmm_node.flood_volume
          mh['user_text_1'] = swmm_node.flooded? ? 'FLOODED' : 'OK'
          mh.write
        end
        
        icm_net.row_objects('cams_pipe').each do |pipe|
          swmm_link = results.link(pipe.id)
          next unless swmm_link
          
          pipe['user_number_1'] = swmm_link.peak_flow
          pipe['user_number_2'] = swmm_link.peak_velocity
          pipe['user_number_3'] = swmm_link.capacity_ratio
          pipe['user_text_1'] = swmm_link.surcharged? ? 'SURCHARGED' : 'OK'
          pipe.write
        end
        
        icm_net.transaction_commit
      end
    end
  end
end
```
