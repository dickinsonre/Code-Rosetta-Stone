```go
// pkg/server/handler.go — Simulation handler

package server

import (
    "encoding/json"
    "net/http"
    "github.com/yourusername/swmm5-go/pkg/engine"
)

func HandleSimulate(w http.ResponseWriter, r *http.Request) {
    // Parse multipart form with .inp file
    file, _, err := r.FormFile("model")
    if err != nil {
        http.Error(w, "Missing model file", 400)
        return
    }
    defer file.Close()
    
    // Read .inp content
    inpContent, _ := io.ReadAll(file)
    
    // Run simulation (concurrent-safe via goroutine)
    resultChan := make(chan *engine.Results)
    go func() {
        sim := engine.NewSimulation(string(inpContent))
        results, err := sim.Run()
        if err != nil {
            resultChan <- &engine.Results{Error: err.Error()}
            return
        }
        resultChan <- results
    }()
    
    // Wait for completion (with timeout)
    select {
    case results := <-resultChan:
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(results)
    case <-time.After(5 * time.Minute):
        http.Error(w, "Simulation timeout", 504)
    }
}

// pkg/engine/engine.go — Main simulation loop

package engine

type Simulation struct {
    project *Project
    started bool
}

func NewSimulation(inpContent string) *Simulation {
    project := ParseINP(inpContent)  // from parser/input.go
    return &Simulation{project: project}
}

func (s *Simulation) Run() (*Results, error) {
    p := s.project
    
    // Initialize
    if err := p.Initialize(); err != nil {
        return nil, err
    }
    TopologicalSort(p)    // from hydraulics/toposort.go
    ComputeInitialFlows(p) // from hydraulics/dwflow.go
    
    results := NewResults(p)
    
    // Time-stepping loop
    for p.CurrentTime.Before(p.EndTime) {
        dt := p.RoutingStep
        
        // Hydrology
        UpdateRainfall(p)           // from hydrology/rain.go
        for _, sc := range p.Subcatchments {
            ComputeRunoff(&sc, dt)   // from hydrology/runoff.go
            ComputeInfiltration(&sc, dt) // from hydrology/infil.go
        }
        
        // Hydraulics
        switch p.RouteModel {
        case DYNWAVE:
            DynwaveRoute(p, dt)      // from hydraulics/routing.go
        case KINWAVE:
            KinwaveRoute(p, dt)      // from hydraulics/kinwave.go
        }
        
        // Controls
        EvaluateControls(p)          // from controls/controls.go
        
        // Mass balance
        UpdateMassBalance(p, dt)     // from output/massbal.go
        
        // Record results at report steps
        if p.IsReportTime() {
            results.Record(p)
        }
        
        p.AdvanceTime(dt)
    }
    
    results.ComputeStatistics()      // from output/stats.go
    return results, nil
}
```

```go
// 1. CONCURRENT MULTI-MODEL SIMULATION
//    Run 100 models simultaneously with goroutines

func BatchSimulate(models []string) []*Results {
    results := make([]*Results, len(models))
    var wg sync.WaitGroup
    
    // Limit concurrency to CPU count
    sem := make(chan struct{}, runtime.NumCPU())
    
    for i, model := range models {
        wg.Add(1)
        go func(idx int, inp string) {
            defer wg.Done()
            sem <- struct{}{}        // acquire
            defer func() { <-sem }() // release
            
            sim := NewSimulation(inp)
            results[idx], _ = sim.Run()
        }(i, model)
    }
    
    wg.Wait()
    return results
}
```

### Key Implementation

```csharp
// Engine/Simulation.cs — Clean C# API

namespace SWMM5;

public class Simulation : IDisposable
{
    private Project _project;
    private bool _running;
    
    public Simulation(string inpFilePath)
    {
        _project = InpParser.Parse(File.ReadAllText(inpFilePath));
    }
    
    public Simulation(Stream inpStream)
    {
        using var reader = new StreamReader(inpStream);
        _project = InpParser.Parse(reader.ReadToEnd());
    }
    
    // Full run — returns all results
    public SimulationResults Run(IProgress<double>? progress = null)
    {
        Initialize();
        var results = new SimulationResults(_project);
        
        while (_project.CurrentTime < _project.EndTime)
        {
            Step();
            
            if (_project.IsReportTime)
                results.Record(_project);
            
            progress?.Report(_project.PercentComplete);
        }
        
        Finalize();
        results.ComputeStatistics();
        return results;
    }
    
    // Step-by-step — for real-time control
    public IEnumerable<SimulationStep> Steps()
    {
        Initialize();
        
        while (_project.CurrentTime < _project.EndTime)
        {
            Step();
            yield return new SimulationStep(
                _project.CurrentTime,
                _project.PercentComplete,
                _project
            );
        }
        
        Finalize();
    }
    
    // Real-time control
    public NodeAccessor Node(string id) => new(_project, id);
    public LinkAccessor Link(string id) => new(_project, id);
    public SubcatchAccessor Subcatchment(string id) => new(_project, id);
    
    private void Step()
    {
        double dt = _project.RoutingStep;
        
        // Hydrology
        RainfallProcessor.Update(_project);
        foreach (var sc in _project.Subcatchments)
        {
            SubcatchmentProcessor.ComputeRunoff(sc, dt);
            Infiltration.Compute(sc, dt);
        }
        
        // Hydraulics
        if (_project.RouteModel == RouteModel.DynWave)
            DynamicWaveRouter.Route(_project, dt);
        else
            KinematicWaveRouter.Route(_project, dt);
        
        // Controls
        RuleEvaluator.Evaluate(_project);
        MassBalance.Update(_project, dt);
        _project.AdvanceTime(dt);
    }
    
    public void Dispose() => _project?.Dispose();
}

// Usage examples:

// 1. Simple run
using var sim = new Simulation("model.inp");

Console.WriteLine($"Peak: {results.Node("Out1").PeakFlow:F2} CFS");

    progressBar.Value = p * 100)));

// 3. Real-time control (PySWMM-equivalent)

{

    else if (sim.Node("J1").Depth < 0.5)
        sim.Link("P1").TargetSetting = 0.0;
}

// 4. LINQ integration (C#-specific superpower)

    .Where(n => n.Value.PeakDepth > n.Value.MaxDepth)
```

### Fortran-Specific: Coarray Parallelism

```fortran
! parallel_mod.f90 — Partition network across coarray images

module parallel_mod
  use iso_fortran_env
  use project_mod
  implicit none

contains

  subroutine partition_network(project, n_partitions)
    type(t_project), intent(inout) :: project
    integer, intent(in) :: n_partitions
    
    integer :: my_image, n_links_per_image, start_link, end_link
    
    my_image = this_image()
    n_links_per_image = project%n_links / n_partitions
    start_link = (my_image - 1) * n_links_per_image + 1
    end_link = min(my_image * n_links_per_image, project%n_links)
    
    ! Each image owns a subset of links
    project%my_links = [(i, i = start_link, end_link)]
  end subroutine

  subroutine parallel_dynwave_step(project, dt)
    type(t_project), intent(inout) :: project
    real(dp), intent(in) :: dt
    
    ! Coarray declarations
    real(dp), codimension[*] :: node_heads(project%n_nodes)
    real(dp), codimension[*] :: link_flows(project%n_links)
    integer :: i
    
    ! Each image computes its links
    do i = project%my_links(1), project%my_links(size(project%my_links))
      call compute_link_flow(project%links(i), project%nodes, dt)
      link_flows(i) = project%links(i)%flow
    end do
    
    ! Synchronize across all images
    sync all
    
    ! Gather all flows (coarray broadcast)
    call co_sum(link_flows)
    
    ! Each image updates its nodes
    do i = project%my_nodes(1), project%my_nodes(size(project%my_nodes))
      call compute_node_depth(project%nodes(i), link_flows, dt)
      node_heads(i) = project%nodes(i)%head
    end do
    
    sync all
    call co_sum(node_heads)
    
  end subroutine

end module

! swmm5_main.f90 — Parallel execution

program swmm5_parallel
  use swmm5_engine_mod
  use parallel_mod
  implicit none
  
  type(t_project) :: project
  
  if (this_image() == 1) then
    print *, "SWMM5 Fortran Engine — Running on", num_images(), "images"
  end if
  
  call parse_inp(project, 'massive_city.inp')
  call partition_network(project, num_images())
  
  ! All images participate in time loop
  do while (project%current_time < project%end_time)
    call parallel_dynwave_step(project, project%routing_step)
    
    ! Only image 1 handles output
    if (this_image() == 1 .and. is_report_time(project)) then
      call write_results(project)
    end if
    
    call advance_time(project)
  end do
  
  if (this_image() == 1) then
    call write_summary(project)
    print *, "Simulation complete. Continuity error:", &
             project%mass_balance%continuity_error, "%"
  end if
  
end program

! Array operations (Fortran's superpower):
! Update ALL node depths in one statement — no loop needed
project%nodes%depth = project%nodes%depth + &
    (dt / project%nodes%area) * &
    (project%nodes%total_inflow - project%nodes%total_outflow)
```

### 8. Mojo — Python Syntax, SIMD Performance

```mojo
# engine.mojo
from algorithm import vectorize
from sys.info import simdwidthof

struct Simulation:
    var project: Project
    
    fn __init__(inout self, inp_content: String):
        self.project = parse_inp(inp_content)
    
    fn step(inout self):
        let dt = self.project.routing_step
        
        # SIMD-accelerated Manning's equation across all links
        alias simd_width = simdwidthof[DType.float64]()
        
        @parameter
        fn compute_friction[width: Int](i: Int):
            let Q = self.project.link_flows.load[width=width](i)
            let A = self.project.link_areas.load[width=width](i)
            let n = self.project.link_roughness.load[width=width](i)
            let R = A / self.project.link_widths.load[width=width](i)
            
            let sf = (n * Q / (A * R ** (2.0/3.0))) ** 2
            self.project.friction_slopes.store[width=width](i, sf)
        
        vectorize[compute_friction, simd_width](self.project.n_links)
    
    fn run(inout self) -> Results:
        self.project.initialize()
        var results = Results(self.project)
        while self.project.current_time < self.project.end_time:
            self.step()
            if self.project.is_report_time():
                results.record(self.project)
        return results
```
