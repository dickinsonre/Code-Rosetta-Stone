### Day 3-4: Engine + Routing

```nim
# src/engine.nim

import types, parser/input, hydraulics/[routing, toposort, dwflow]
import hydrology/[subcatch, infil, rain]
import output/massbal

type
  Simulation* = object
    project*: Project
    stepCount: int

  Results* = object
    times*: seq[float64]
    nodeDepths*: seq[seq[float64]]
    linkFlows*: seq[seq[float64]]
    continuityError*: float64

proc newSimulation*(inpFile: string): Simulation =
  result.project = parseInp(readFile(inpFile))
  result.stepCount = 0

proc step(sim: var Simulation) =
  let dt = sim.project.routingStep

  # Hydrology
  updateRainfall(sim.project)
  for sc in sim.project.subcatchments.mitems:
    computeRunoff(sc, dt)
    computeInfiltration(sc, dt)

  # Hydraulics
  case sim.project.routeModel
  of rmDynWave:
    dynwaveStep(sim.project, dt)
  of rmKinWave:
    kinwaveRoute(sim.project, dt)
  of rmSteady:
    discard

  # Controls + mass balance
  updateMassBalance(sim.project, dt)
  sim.project.currentTime += dt
  sim.stepCount.inc

proc run*(sim: var Simulation): Results =
  ## Run complete simulation, return results
  sim.project.validate()
  topoSort(sim.project)
  computeInitialFlows(sim.project)

  result = Results()
  let reportRatio = (sim.project.reportStep / sim.project.routingStep).int

  while sim.project.currentTime < sim.project.endTime:
    sim.step()

    if sim.stepCount mod reportRatio == 0:
      result.times.add(sim.project.currentTime)
      result.nodeDepths.add(sim.project.nodes.mapIt(it.depth))
      result.linkFlows.add(sim.project.links.mapIt(it.flow))

  result.continuityError = computeContinuityError(sim.project)

# Step-by-step iteration (PySWMM-style)
iterator steps*(sim: var Simulation): Simulation =
  sim.project.validate()
  topoSort(sim.project)
  computeInitialFlows(sim.project)

  while sim.project.currentTime < sim.project.endTime:
    sim.step()
    yield sim
```
```nim
# src/hydraulics/routing.nim — Saint-Venant in Nim

import math, ../types

const Gravity* = 32.2  # ft/s²

proc frictionSlope*(link: Link, flow, area: float64): float64 =
  ## Manning's equation for friction slope
  if area <= 0.0 or flow == 0.0: return 0.0

  let hRadius = area / link.xsect.wMax  # simplified
  if hRadius <= 0.0: return 0.0

  let sf = link.roughness * flow / (area * pow(hRadius, 2.0/3.0))
  result = sf * sf
  if flow < 0.0: result = -result

proc computeFlow*(link: Link, nodes: seq[Node], dt: float64): float64 =
  ## Explicit finite-difference Saint-Venant momentum
  let area = link.xsect.aFull * (link.depth / link.xsect.geom1).clamp(0.0, 1.0)
  if area <= 0.0: return 0.0

  let
    hUp = nodes[link.fromNode].head
    hDown = nodes[link.toNode].head
    dhdx = (hUp - hDown) / link.length
    sf = frictionSlope(link, link.oldFlow, area)

  result = link.oldFlow + dt * Gravity * area * (dhdx - sf)

  # Velocity cap
  let qMax = area * 50.0
  result = result.clamp(-qMax, qMax)

proc dynwaveStep*(project: var Project, dt: float64) =
  ## Route all links for one time step

  # Save old state
  for link in project.links.mitems:
    link.oldFlow = link.flow
    link.oldDepth = link.depth

  # Compute new flows
  for link in project.links.mitems:
    if link.linkType == ltConduit:
      link.flow = computeFlow(link, project.nodes, dt)

  # Update node depths
  for node in project.nodes.mitems:
    if node.nodeType == ntOutfall: continue
    var totalIn = node.lateralInflow
    var totalOut = 0.0
    for i in node.inlinks: totalIn += project.links[i].flow.abs
    for i in node.outlinks: totalOut += project.links[i].flow.abs
    node.totalInflow = totalIn

    let dv = (totalIn - totalOut) * dt
    node.volume = max(node.volume + dv, 0.0)
    node.depth = min(node.volume / max(node.maxDepth, 1e-6), node.maxDepth)
    node.head = node.invertElev + node.depth
    node.overflow = if node.depth >= node.maxDepth:
      max(totalIn - totalOut, 0.0) else: 0.0
```

### The Key Code

```pascal
// SWMM5Engine.pas — Full circle: Delphi engine + Delphi GUI

unit SWMM5Engine;

interface

uses
  SysUtils, Classes, Math,
  SWMM5Types, SWMM5Parser, SWMM5Routing, SWMM5Subcatch,
  SWMM5Infil, SWMM5Rain, SWMM5MassBal, SWMM5TopoSort;

type
  TSWMM5Simulation = class
  private
    FProject: TProject;
    FStepCount: Integer;
    procedure DoStep;
  public
    constructor Create(const InpFile: string);
    destructor Destroy; override;
    function Run: TSWMM5Results;
    function NodeDepth(const ID: string): Double;
    function LinkFlow(const ID: string): Double;
    property PercentComplete: Double read GetPercentComplete;
  end;

implementation

constructor TSWMM5Simulation.Create(const InpFile: string);
var
  InpContent: TStringList;
begin
  inherited Create;
  InpContent := TStringList.Create;
  try
    InpContent.LoadFromFile(InpFile);
    FProject := ParseINP(InpContent.Text);
  finally
    InpContent.Free;
  end;
  FStepCount := 0;
end;

function TSWMM5Simulation.Run: TSWMM5Results;
var
  dt: Double;
begin
  FProject.Validate;
  TopologicalSort(FProject);
  ComputeInitialFlows(FProject);

  Result := TSWMM5Results.Create(FProject);
  dt := FProject.RoutingStep;

  while FProject.CurrentTime < FProject.EndTime do
  begin
    DoStep;

    if FStepCount mod FProject.ReportRatio = 0 then
      Result.Record(FProject);

    FProject.CurrentTime := FProject.CurrentTime + dt;
  end;

  Result.ComputeStatistics;
end;

procedure TSWMM5Simulation.DoStep;
var
  dt: Double;
  I: Integer;
begin
  dt := FProject.RoutingStep;

  // Hydrology
  UpdateRainfall(FProject);
  for I := 0 to FProject.SubcatchCount - 1 do
  begin
    ComputeRunoff(FProject.Subcatchments[I], dt);
    ComputeInfiltration(FProject.Subcatchments[I], dt);
  end;

  // Hydraulics — Saint-Venant solver
  case FProject.RouteModel of
    rmDynWave: DynWaveRoute(FProject, dt);
    rmKinWave: KinWaveRoute(FProject, dt);
  end;

  // Mass balance
  UpdateMassBalance(FProject, dt);
  Inc(FStepCount);
end;

end.
```

### The R Engine

```r
# R/engine.R

#' Run a SWMM5 simulation
#' @param inp_file Path to SWMM5 .inp file
#' @param ... Optional parameter overrides (manning_n, imperv, etc.)
#' @return A swmm5_results object (inherits from data.frame)
#' @export
#' @examples
#' results <- swmm5_run("model.inp")
#' plot(results)
#' results <- swmm5_run("model.inp", manning_n = 0.015)
swmm5_run <- function(inp_file, ...) {
  overrides <- list(...)

  # Parse
  project <- swmm5_parse(inp_file)

  # Apply overrides
  if (!is.null(overrides$manning_n)) {
    project$links$roughness <- overrides$manning_n
  }
  if (!is.null(overrides$imperv)) {
    project$subcatchments$imperv <- overrides$imperv
  }

  # Initialize
  project <- swmm5_init(project)
  project <- topo_sort(project)
  project <- compute_initial_flows(project)

  # Pre-allocate results
  n_steps <- ceiling((project$end_time - project$start_time) / project$report_step)
  results <- data.frame(
    time = numeric(n_steps),
    matrix(0, nrow = n_steps, ncol = length(project$nodes),
           dimnames = list(NULL, paste0(project$nodes$id, "_depth"))),
    matrix(0, nrow = n_steps, ncol = length(project$links),
           dimnames = list(NULL, paste0(project$links$id, "_flow")))
  )

  # Time-stepping loop
  step_count <- 0L
  report_idx <- 0L
  dt <- project$routing_step

  while (project$current_time < project$end_time) {
    # Hydrology
    project <- update_rainfall(project)
    for (i in seq_along(project$subcatchments$id)) {
      project <- compute_runoff(project, i, dt)
      project <- compute_infiltration(project, i, dt)
    }

    # Hydraulics
    if (project$route_model == "DYNWAVE") {
      project <- dynwave_step(project, dt)
    } else {
      project <- kinwave_step(project, dt)
    }

    # Mass balance
    project <- update_mass_balance(project, dt)

    # Record at report steps
    step_count <- step_count + 1L
    if (step_count %% project$report_ratio == 0L) {
      report_idx <- report_idx + 1L
      results$time[report_idx] <- project$current_time
      for (j in seq_along(project$nodes$id)) {
        results[report_idx, j + 1] <- project$nodes$depth[j]
      }
    }

    project$current_time <- project$current_time + dt
  }

  # Trim and class
  results <- results[1:report_idx, ]
  class(results) <- c("swmm5_results", "data.frame")
  attr(results, "project") <- project
  attr(results, "continuity_error") <- compute_continuity_error(project)

  results
}

#' @export
plot.swmm5_results <- function(x, nodes = NULL, ...) {
  if (is.null(nodes)) {
    # Plot first 5 nodes
    depth_cols <- grep("_depth$", names(x), value = TRUE)[1:5]
  } else {
    depth_cols <- paste0(nodes, "_depth")
  }

  matplot(x$time, x[, depth_cols], type = "l", lty = 1,
          xlab = "Time (hr)", ylab = "Depth (m)",
          main = "SWMM5 Node Depths", ...)
  legend("topright", legend = gsub("_depth", "", depth_cols),
         col = 1:length(depth_cols), lty = 1, cex = 0.8)
}
```

```kotlin
class Simulation(inpContent: String) {

    suspend fun run(onProgress: (Double) -> Unit = {}): Results {
        // Kotlin coroutines = non-blocking on Android

            val results = Results(project)
            while (project.currentTime < project.endTime) {

            }
            results
        }
    }
```

### Engine 7: Mojo (3-4 Weeks)

```mojo
# SIMD-accelerated Manning's equation
# Processes 4-8 conduits per CPU instruction

from algorithm import vectorize
from sys.info import simdwidthof

alias simd_width = simdwidthof[DType.float64]()

fn dynwave_step(inout project: Project, dt: Float64):
    let n = project.n_links

    # SIMD: compute ALL friction slopes simultaneously
    @parameter
    fn compute_sf[width: Int](i: Int):
        let Q = project.flows.load[width=width](i)
        let A = project.areas.load[width=width](i)
        let n_val = project.roughness.load[width=width](i)
        let R = project.h_radius.load[width=width](i)
        let sf = (n_val * Q / (A * R ** (2.0/3.0))) ** 2
        project.sf.store[width=width](i, sf)

    vectorize[compute_sf, simd_width](n)
    # ↑ This one line processes 4-8 conduits per CPU cycle
    # 100-500× faster than scalar Python
```
