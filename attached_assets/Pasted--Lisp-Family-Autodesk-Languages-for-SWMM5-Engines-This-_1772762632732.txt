```lisp
(defun swmm5:make-node (id invert-elev max-depth
                         / node)
  "Create a SWMM5 node as an association list"
  (list
    (cons 'id id)
    (cons 'type 'junction)
    (cons 'invert-elev invert-elev)
    (cons 'max-depth max-depth)
    ;; Dynamic state
    (cons 'depth 0.0)
    (cons 'head invert-elev)
    (cons 'volume 0.0)
    (cons 'lateral-inflow 0.0)
    (cons 'total-inflow 0.0)
    (cons 'overflow 0.0)
    ;; Connectivity
    (cons 'inlinks nil)
    (cons 'outlinks nil)
  )
)

(defun swmm5:make-link (id from-node to-node length
                         roughness diameter / link)
  "Create a SWMM5 conduit as an association list"
  (list
    (cons 'id id)
    (cons 'type 'conduit)
    (cons 'from-node from-node)
    (cons 'to-node to-node)
    (cons 'length length)
    (cons 'roughness roughness)
    (cons 'diameter diameter)
    (cons 'a-full (* pi (/ (* diameter diameter) 4.0)))
    ;; Dynamic state
    (cons 'flow 0.0)
    (cons 'depth 0.0)
    (cons 'velocity 0.0)
    (cons 'old-flow 0.0)
    (cons 'setting 1.0)
  )
)
```

```lisp
;;; hydraulics/routing.lsp — Saint-Venant equations in AutoLISP
;;; The same algorithm, in the oldest high-level language family

(setq *GRAVITY* 32.2)  ; ft/s² (9.81 for metric)

(defun swmm5:friction-slope (link flow area / h-radius sf)
  "Compute Manning's friction slope for a conduit"
  (if (or (<= area 0.0) (= flow 0.0))
    0.0
    (progn
      (setq h-radius (/ area (swmm5:get link 'diameter)))
      (if (<= h-radius 0.0)
        0.0
        (progn
          (setq sf (/ (* (swmm5:get link 'roughness) flow)
                      (* area (expt h-radius (/ 2.0 3.0)))))
          (setq sf (* sf sf))
          (if (< flow 0.0)
            (- sf)
            sf
          )
        )
      )
    )
  )
)

(defun swmm5:compute-flow (link nodes dt
                            / area h-up h-down dhdx sf q-new q-max
                              from-idx to-idx)
  "Compute new conduit flow using Saint-Venant momentum equation"
  (setq area (swmm5:get link 'a-full))  ; simplified: use full area
  (if (<= area 0.0)
    0.0
    (progn
      (setq from-idx (swmm5:get link 'from-node))
      (setq to-idx   (swmm5:get link 'to-node))
      (setq h-up   (swmm5:get (nth from-idx nodes) 'head))
      (setq h-down (swmm5:get (nth to-idx nodes) 'head))
      (setq dhdx (/ (- h-up h-down)
                    (swmm5:get link 'length)))
      (setq sf (swmm5:friction-slope link
                 (swmm5:get link 'old-flow) area))
      
      ;; Explicit finite-difference: Saint-Venant momentum
      (setq q-new (+ (swmm5:get link 'old-flow)
                     (* dt *GRAVITY* area (- dhdx sf))))
      
      ;; Velocity cap
      (setq q-max (* area 50.0))
      (max (- q-max) (min q-new q-max))
    )
  )
)

(defun swmm5:dynwave-step (project dt / links nodes i link)
  "Route all conduits for one dynamic wave time step"
  (setq links (swmm5:get project 'links))
  (setq nodes (swmm5:get project 'nodes))
  
  ;; Save old flows
  (setq i 0)
  (repeat (length links)
    (setq link (nth i links))
    (nth-set links i
      (swmm5:set link 'old-flow (swmm5:get link 'flow)))
    (setq i (1+ i))
  )
  
  ;; Compute new flows for all conduits
  (setq i 0)
  (repeat (length links)
    (setq link (nth i links))
    (if (eq (swmm5:get link 'type) 'conduit)
      (nth-set links i
        (swmm5:set link 'flow
          (swmm5:compute-flow link nodes dt)))
    )
    (setq i (1+ i))
  )
  
  ;; Update node depths
  (swmm5:update-node-depths project dt)
  
  ;; Return updated project
  (swmm5:set
    (swmm5:set project 'links links)
    'nodes (swmm5:get project 'nodes))
)
```

```lisp
;;; swmm5-engine.lsp — Main simulation loop

(defun swmm5:run (inp-file / project dt step-count results)
  "Run a complete SWMM5 simulation from an .inp file"
  (princ "\nSWMM5 AutoLISP Engine v5.2.4")
  (princ (strcat "\nLoading: " inp-file))
  
  ;; Parse
  (setq project (swmm5:parse-inp inp-file))
  (princ (strcat "\nNodes: " (itoa (length (swmm5:get project 'nodes)))))
  (princ (strcat "\nLinks: " (itoa (length (swmm5:get project 'links)))))
  
  ;; Initialize
  (setq project (swmm5:topo-sort project))
  (setq project (swmm5:init-flows project))
  
  (setq dt (swmm5:get project 'routing-step))
  (setq step-count 0)
  (setq results nil)
  
  ;; Time-stepping loop
  (princ "\nRunning simulation...")
  (while (< (swmm5:get project 'current-time)
            (swmm5:get project 'end-time))
    
    ;; Hydrology
    (setq project (swmm5:update-rainfall project))
    (setq project (swmm5:compute-runoff project dt))
    
    ;; Hydraulics — Saint-Venant solver
    (setq project (swmm5:dynwave-step project dt))
    
    ;; Mass balance
    (setq project (swmm5:update-mass-balance project dt))
    
    ;; Record results at report steps
    (setq step-count (1+ step-count))
    (if (= (rem step-count (swmm5:get project 'report-ratio)) 0)
      (setq results (cons (swmm5:snapshot project) results))
    )
    
    ;; Advance time
    (setq project
      (swmm5:set project 'current-time
        (+ (swmm5:get project 'current-time) dt)))
    
    ;; Progress (every 100 steps)
    (if (= (rem step-count 100) 0)
      (princ (strcat "\r  " 
        (rtos (* 100.0 (/ (- (swmm5:get project 'current-time)
                             (swmm5:get project 'start-time))
                          (- (swmm5:get project 'end-time)
                             (swmm5:get project 'start-time))))
               2 1) "%"))
    )
  )
  
  (princ "\nSimulation complete.")
  (princ (strcat "\nContinuity error: "
    (rtos (swmm5:continuity-error project) 2 3) "%"))
  
  ;; Return results
  (reverse results)
)
```

```lisp
(defun swmm5:c3d-extract-network (/ ss i ent etype
                                     nodes links)
  "Extract pipe network elements from Civil 3D drawing"
  (setq nodes nil links nil)
  
  ;; Get all Civil 3D pipe network structures (manholes)
  (setq ss (ssget "X" '((0 . "AECC_STRUCTURE"))))
  (if ss
    (progn
      (setq i 0)
      (repeat (sslength ss)
        (setq ent (ssname ss i))
        (setq nodes (cons
          (swmm5:make-node
            (cdr (assoc 2 (entget ent)))          ; Structure name
            (cdr (assoc 10 (entget ent)))         ; Sump elevation
            (- (cdr (assoc 38 (entget ent)))      ; Rim - sump = depth
               (cdr (assoc 10 (entget ent))))
          )
          nodes))
        (setq i (1+ i))
      )
    )
  )
  
  ;; Get all Civil 3D pipe network pipes
  (setq ss (ssget "X" '((0 . "AECC_PIPE"))))
  (if ss
    (progn
      (setq i 0)
      (repeat (sslength ss)
        (setq ent (ssname ss i))
        (setq links (cons
          (swmm5:make-link
            (cdr (assoc 2 (entget ent)))          ; Pipe name
            (cdr (assoc 330 (entget ent)))        ; Start structure
            (cdr (assoc 331 (entget ent)))        ; End structure
            (cdr (assoc 42 (entget ent)))         ; Length
            0.013                                  ; Default Manning's n
            (/ (cdr (assoc 40 (entget ent))) 12.0) ; Inner diameter (in→ft)
          )
          links))
        (setq i (1+ i))
      )
    )
  )
  
  (if (and nodes links)
    (list (cons 'nodes (reverse nodes))
          (cons 'links (reverse links)))
    nil
  )
)

(defun swmm5:c3d-display-results (network results / i node link
                                    depth-ratio flow-ratio color)
  "Color Civil 3D pipes and structures by SWMM5 results"
  
  (princ "\nColoring network by simulation results...")
  
  ;; Color structures by depth ratio (depth / max_depth)
  (foreach node-result (swmm5:get (car (last results)) 'node-results)
    (setq depth-ratio (/ (swmm5:get node-result 'peak-depth)
                         (max (swmm5:get node-result 'max-depth) 0.01)))
    (setq color
      (cond
        ((> depth-ratio 1.0) 1)    ; RED — flooded!
        ((> depth-ratio 0.8) 30)   ; ORANGE — near capacity
        ((> depth-ratio 0.5) 2)    ; YELLOW — moderate
        (t                   3)    ; GREEN — OK
      )
    )
    ;; Change entity color
    (swmm5:set-entity-color
      (swmm5:find-structure (swmm5:get node-result 'id))
      color)
  )
  
  ;; Color pipes by capacity ratio (flow / full_flow)
  (foreach link-result (swmm5:get (car (last results)) 'link-results)
    (setq flow-ratio (swmm5:get link-result 'capacity-ratio))
    (setq color
      (cond
        ((> flow-ratio 1.0)  1)   ; RED — surcharged
        ((> flow-ratio 0.8) 30)   ; ORANGE
        ((> flow-ratio 0.5)  2)   ; YELLOW
        (t                   3)   ; GREEN
      )
    )
    (swmm5:set-entity-color
      (swmm5:find-pipe (swmm5:get link-result 'id))
      color)
    
    ;; Add text label with peak flow
    (swmm5:add-result-label
      (swmm5:find-pipe (swmm5:get link-result 'id))
      (strcat (rtos (swmm5:get link-result 'peak-flow) 2 1) " CFS"))
  )
  
  ;; Regenerate display
  (command "._REGEN")
  
  (princ (strcat "\n  " 
    (itoa (length (swmm5:get (car (last results)) 'flooded-nodes)))
    " flooded structures highlighted in RED"))
)
```

```lisp
(defpackage :swmm5
  (:use :cl)
  (:export :run-simulation :make-simulation :node-depth :link-flow))

(in-package :swmm5)

(defstruct node
  (id "" :type string)
  (invert-elev 0.0d0 :type double-float)
  (max-depth 0.0d0 :type double-float)
  (depth 0.0d0 :type double-float)
  (head 0.0d0 :type double-float)
  (volume 0.0d0 :type double-float)
  (total-inflow 0.0d0 :type double-float)
  (overflow 0.0d0 :type double-float))

(defstruct link
  (id "" :type string)
  (from-node 0 :type fixnum)
  (to-node 0 :type fixnum)
  (length 0.0d0 :type double-float)
  (roughness 0.013d0 :type double-float)
  (diameter 0.0d0 :type double-float)
  (a-full 0.0d0 :type double-float)
  (flow 0.0d0 :type double-float)
  (old-flow 0.0d0 :type double-float))

(defconstant +gravity+ 32.2d0)

;; Type declarations → SBCL compiles to native machine code
(declaim (ftype (function (link double-float double-float) double-float)
                friction-slope))

(defun friction-slope (link flow area)
  (declare (optimize (speed 3) (safety 0))
           (type double-float flow area))
  (if (or (<= area 0.0d0) (= flow 0.0d0))
    0.0d0
    (let* ((h-radius (/ area (link-diameter link)))
           (sf (/ (* (link-roughness link) flow)
                  (* area (expt h-radius (/ 2.0d0 3.0d0))))))
      (declare (type double-float h-radius sf))
      (setf sf (* sf sf))
      (if (< flow 0.0d0) (- sf) sf))))

(defun compute-flow (link nodes dt)
  (declare (optimize (speed 3) (safety 0))
           (type double-float dt))
  (let* ((area (link-a-full link))
         (h-up (node-head (aref nodes (link-from-node link))))
         (h-down (node-head (aref nodes (link-to-node link))))
         (dhdx (/ (- h-up h-down) (link-length link)))
         (sf (friction-slope link (link-old-flow link) area))
         (q-new (+ (link-old-flow link)
                   (* dt +gravity+ area (- dhdx sf))))
         (q-max (* area 50.0d0)))
    (max (- q-max) (min q-new q-max))))
```

### Clojure — JVM Concurrent Engine

```clojure
;; swmm5/engine.clj — Functional SWMM5 on the JVM
;; Immutable data + parallel map = thread-safe multi-model runs

(ns swmm5.engine
  (:require [swmm5.parser :as parser]
            [swmm5.routing :as routing]
            [swmm5.hydrology :as hydro]))

(def ^:const GRAVITY 32.2)

(defn friction-slope [{:keys [roughness]} flow area]
  (if (or (<= area 0.0) (zero? flow))
    0.0
    (let [h-radius (/ area (:diameter link))
          sf (/ (* roughness flow)
                (* area (Math/pow h-radius (/ 2.0 3.0))))]
      (* (if (neg? flow) -1.0 1.0) (* sf sf)))))

(defn step
  "Pure function: project → project (immutable state update)"
  [project dt]
  (-> project
      (hydro/update-rainfall)
      (hydro/compute-runoff dt)
      (routing/dynwave-step dt)
      (update :current-time + dt)))

(defn run-simulation [inp-file]
  (let [project (parser/parse-inp (slurp inp-file))
        project (routing/initialize project)]
    (loop [p project, results []]
      (if (>= (:current-time p) (:end-time p))
        results
        (let [p' (step p (:routing-step p))]
          (recur p' (conj results (snapshot p'))))))))

;; THE CLOJURE SUPERPOWER: parallel parameter sweep
;; pmap = parallel map across all CPU cores
(defn sensitivity-analysis [inp-file param-values]
  (pmap (fn [n]
    (let [results (run-simulation inp-file {:manning-n n})]
      {:n n :peak-flow (peak-flow results "Out1")}))
    param-values))

;; 10 models in parallel, immutable state, zero race conditions:
```

```scheme
;;; swmm5.rkt — Racket (modern Scheme)
;;; Elegant, minimal, perfect for teaching

#lang racket

(struct node (id invert-elev max-depth depth head volume) #:mutable)
(struct link (id from to length roughness diameter flow old-flow) #:mutable)

(define GRAVITY 32.2)

(define (friction-slope lnk flow area)
  (cond
    [(<= area 0.0) 0.0]
    [(= flow 0.0) 0.0]
    [else
     (let* ([h-radius (/ area (link-diameter lnk))]
            [sf (/ (* (link-roughness lnk) flow)
                   (* area (expt h-radius 2/3)))])
       (* (if (negative? flow) -1 1)
          (* sf sf)))]))

;; Pattern matching for cross-section geometry
(define (area-at-depth shape depth)
  (match shape
    [(list 'circular d)
     (let ([θ (* 2 (acos (- 1 (/ (* 2 (min depth d)) d))))])
       (* (/ (* d d) 4) (- θ (* (sin θ) (cos θ)))))]
    [(list 'rectangular h w)
     (* (min depth h) w)]
    [_ 0.0]))

;; Simulation as a fold (functional elegance)
(define (run-simulation project)
  (for/fold ([p project]
             [results '()])
            ([_ (in-range (total-steps p))])
    (let ([p* (step p (project-dt p))])
      (values p* (cons (snapshot p*) results)))))

;; Run: racket swmm5.rkt model.inp
;; Teaching value: 
;;   "This is the Saint-Venant equation. In 20 lines of Scheme."
```

```python
# Dynamo visual scripting node for Civil 3D

# Input: Civil 3D Pipe Network (from Dynamo Civil 3D toolkit)

import clr
clr.AddReference('AeccDbMgd')  # Civil 3D API
from Autodesk.Civil.DatabaseServices import *

def extract_network(pipe_network):
    """Dynamo node: Extract SWMM5 data from Civil 3D pipe network"""
    nodes = []
    links = []
    
    for structure in pipe_network.GetStructureIds():
        obj = structure.GetObject()
        nodes.append({
            'id': obj.Name,
            'invert': obj.SumpElevation,
            'rim': obj.RimElevation,
            'x': obj.Position.X,
            'y': obj.Position.Y
        })

        obj = pipe.GetObject()
        links.append({
            'id': obj.Name,
            'from': obj.StartStructureName,
            'to': obj.EndStructureName,
            'length': obj.Length2D,
            'diameter': obj.InnerDiameterOrWidth / 12.0,
            'slope': obj.Slope
        })
```
